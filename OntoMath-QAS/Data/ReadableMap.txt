SELECT ?name ?comment
WHERE {{
    ?target rdfs:label ?name.
    ?target rdfs:comment ?comment.
    ?target rdfs:subClassOf ?parent.
    ?parent rdfs:label ?label.
    filter(?label = ""{0}""@ru)
    filter langMatches(lang(?name),""ru"")
}}
LIMIT 10
